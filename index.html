contract Ballot {

	struct Proposal {
		uint voteCount;
	}

	constructor(uint8 _numProposals) public {
		proposals.length = _numProposals;
	}

	function vote(uint8 toProposal) public {
		proposals[toProposal].voteCount ++;
	}
}
